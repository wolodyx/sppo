void* func(void* arg)
{
  while(1)
  {
    sleep(1);
    fprintf(stderr, "A");
  }
  return NULL;
}

int main(int argc, char** argv)
{
  pthread_t thread;
  if(pthread_create(&thread, NULL, &func, NULL)!=0)
  {
    fprintf(stderr, "Error thread creating\n");
    return 1;
  }
  printf("thread id = %d\n", (int)thread);
  while(1)
  {
    sleep(1);
    fprintf(stderr, "B");
  }
  return 0;
}
